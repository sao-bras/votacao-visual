}); document.getElementById('setup-area').classList.add('hidden'); document.getElementById('voting-area').classList.remove('hidden'); document.getElementById('title').innerText = "Escolha seu Líder"; } function registrarVoto(nome) { const index = candidatos.findIndex(c => c.nome === nome); candidatos[index].votos++; salvarNoCache(); // Salva o voto imediatamente no cache document.getElementById('voting-area').classList.add('hidden'); } } function exibirResultados() { const resDiv = document.getElementById('final-results'); resDiv.innerHTML = ""; // Criar uma cópia para ordenar sem bagunçar o cache original se quiserem continuar let ordenados = [...candidatos].sort((a, b) => b.votos - a.votos);
